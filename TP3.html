# Mtodo para o clculo de mnimos quadrados
def getMQ(b, m):
    mq = 0
    for p,a in zip(peso, altura):
        mq += ( a - (m*p+b))**2
    return mq

mq = getMQ(b, m)
print("O valor dos mnimos quadrados : {}".format(np.round(mq, 4)))

5° Calculando o novo valor de b¶

Para calcular o novo valor de b vamos precisar definir o learning rate e o step size. O learnin rate um valor que multiplica a derivada da funo mltiplos quadrados. Se o lr for muito pequeno ou muito grande no vamos ser capazes de achar o tamanho adequado de step size. O step size o quanto o valor de b - ou m, mas nesse caso b - vai variar.

In [11]:
def setnewb(d_mq, lr):
    # Learning Rate
    if lr == 0:
        lr = 0.01

    # Step Size
    ss = d_mq * lr

    # O novo valor de b  o antigo valor de b menos o step size
    return b - ss

b = setnewb(derivada_mq_em_relacao_a_b, 0.1)

# J criamos a funo para calcular o novo valor de b, nos resta agora criar a funo para o novo valor de m
def setnewm(d_mq_m, lr):
    # d_mq_m  a derivada de MQ em relao  m
    
    # Learning Rate
    if lr == 0:
        lr = 0.01

    # Step Size
    ss = d_mq_m * lr

    # O novo valor de b  o antigo valor de b menos o step size
    return m - ss

b = setnewb(derivada_mq_em_relacao_a_b, 0.00001)
m = setnewm(derivada_mq_em_relacao_a_m, 0.00001)

print("Os novos valores de b e m so respectivamente: {} e {}".format(b, m))
